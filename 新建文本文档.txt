export class AssembleScene extends Base2DScene {
    viewModel: ViewModel;
    edgesWidth = 6;
    r = 6;
    tipA: Mesh;
    tiptextA: GUI.TextBlock;

    tipC: Mesh;
    tiptextC: GUI.TextBlock;

    tipB: Mesh;
    tiptextB: GUI.TextBlock;

    tipD: Mesh;
    tiptextD: GUI.TextBlock;

    tipP1: Mesh;
    tiptextP1: GUI.TextBlock;

    tipP2: Mesh;
    tiptextP2: GUI.TextBlock;

    tipP3: Mesh;
    tiptextP3: GUI.TextBlock;

    tipP4: Mesh;
    tiptextP4: GUI.TextBlock;


    Circle: LinesMesh;
    Ellipse: LinesMesh;

    LineA: LinesMesh;
    LineB: LinesMesh;
    LineC: LinesMesh;
    LineD: LinesMesh;
    a = 5;
    b = 4;
    length = 0.5;
    IsMoved = true;
    constructor(vm: Vue) {
        super();
        this.viewModel = vm as ViewModel;
        this.init();
    }
    /** 
     * 눗왯넹당路零 
     */
    resize() {
        super.resize();
        this.changeCameraSize();
    }
    /**
     * 놓迦뺏貢목
     * @param scene 
     */
    initMesh(scene: BABYLON.Scene) {
        this.tipP1 = Mesh.CreateSphere('ti1', 8, 1.5, scene);
        this.tipA = Mesh.CreateSphere('tipA', 8, 1.5, scene);
        this.tipB = Mesh.CreateSphere('tipB', 8, 1.5, scene);

        this.tipP2 = new Mesh('ti');
        this.tipP3 = new Mesh('ti');
        this.tipP4 = new Mesh('ti');
        this.tipC = new Mesh('ti');
        this.tipD = new Mesh('ti');
        this.tipA.isVisible = this.tipB.isVisible = this.tipP1.isVisible = false;
    }

    /**
     * 놓迦뺏鑒令
     * @param engine 
     */
    initValue(scene: BABYLON.Scene): void {
        this.initMesh(scene);
        const advancedTexture = GUI.AdvancedDynamicTexture.CreateFullscreenUI('UI');
        CoordinateSystem.Create2DSystem(15, 1, {
            height: 20, width: 30, color: '#000000',
            fontSize: 12, fontFamily: '', fontStyle: ''
        }, this.edgesWidth, advancedTexture, scene);

        const options = {
            height: '30px', width: '30px', color: Color.StringBlack,
            fontSize: '24px', fontFamily: 'Times New Roman', fontStyle: 'italic'
        };

        this.tiptextA = LabelUtils.CreateLabelWithOffset(advancedTexture, this.tipA, 'A', '20px', '25px', options);
        this.tiptextB = LabelUtils.CreateLabelWithOffset(advancedTexture, this.tipB, 'B', '20px', '25px', options);
        this.tiptextC = LabelUtils.CreateLabelWithOffset(advancedTexture, this.tipC, 'C', '20px', '25px', options);
        this.tiptextD = LabelUtils.CreateLabelWithOffset(advancedTexture, this.tipD, 'D', '20px', '25px', options);
        const options2 = {
            height: '30px', width: '100px', color: Color.StringBlack,
            fontSize: '24px', fontFamily: 'Times New Roman', fontStyle: 'italic'
        };

        this.tiptextP1 = LabelUtils.CreateLabelWithOffset(advancedTexture, this.tipP1, 'P (-a,b)', '60px', '25px', options2);
        this.tiptextP2 = LabelUtils.CreateLabelWithOffset(advancedTexture, this.tipP2, 'P (a,b)', '60px', '25px', options2);
        this.tiptextP3 = LabelUtils.CreateLabelWithOffset(advancedTexture, this.tipP3, 'P (-a,-b)', '60px', '25px', options2);
        this.tiptextP4 = LabelUtils.CreateLabelWithOffset(advancedTexture, this.tipP4, 'P (a,-b)', '60px', '25px', options2);

        const ImageLabelOption = { height: '40px', width: '40px', color: '#FFFFFF' };
        LabelUtils.CreateImageLabel(advancedTexture, this.tipA, `${dot}`, ImageLabelOption);
        LabelUtils.CreateImageLabel(advancedTexture, this.tipB, `${dot}`, ImageLabelOption);
        LabelUtils.CreateImageLabel(advancedTexture, this.tipP1, `${dot}`, ImageLabelOption);

        LabelUtils.CreateImageLabel(advancedTexture, this.tipC, `${dot2}`, ImageLabelOption);
        LabelUtils.CreateImageLabel(advancedTexture, this.tipD, `${dot2}`, ImageLabelOption);

        this.tipP1.position = new Vector3(-this.a, this.b, 0);

        this.Circle = FormulaLineUtils.CreateUpdateCircle(this.r, 360, Color.Blue, this.edgesWidth, this.Circle, scene);
        this.Ellipse = FormulaLineUtils.CreateUpdateEllipse(this.a, this.b, Color.Blue, this.edgesWidth, this.Ellipse, scene);
        this.LineA = LinesBuild.CreateUpdateLines([Vector3.Zero(), Vector3.Zero()], Color.Black, this.edgesWidth, this.LineA, scene);
        this.LineB = LinesBuild.CreateUpdateLines([Vector3.Zero(), Vector3.Zero()], Color.Black, this.edgesWidth, this.LineB, scene);
        this.LineC = LinesBuild.CreateUpdateLines([Vector3.Zero(), Vector3.Zero()], Color.Black, this.edgesWidth, this.LineC, scene);
        this.LineD = LinesBuild.CreateUpdateLines([Vector3.Zero(), Vector3.Zero()], Color.Black, this.edgesWidth, this.LineD, scene);
    }

    /**
     * 눼쉔끝쒼
     * @param engine 
     */
    createScene(engine: BABYLON.Engine): BABYLON.Scene {
        const canvas = engine.getRenderingCanvas();
        const scene = new BABYLON.Scene(engine);
        scene.clearColor.set(1, 1, 1, 1);
        this.createTargetCamera4Math(scene, 18);
        this.initValue(scene);
        this.addPointerEventListener(canvas, scene);
        this.reset();
        return scene;
    }
    /**
     * 癎覺솰桂
     * @param canvas 
     * @param scene 
     */
    addPointerEventListener(canvas: HTMLCanvasElement, scene: BABYLON.Scene): void {
        let startingPoint: BABYLON.Vector3;
        let currentMesh: BABYLON.AbstractMesh;
        const plan = BABYLON.Mesh.CreatePlane('plan', 160, scene);
        plan.isVisible = false;
        const getGroundPosition = (evt: any) => {
            const pickinfo = scene.pick(scene.pointerX, scene.pointerY, function (mesh) {
                return mesh === plan;
            });
            if (pickinfo.hit) {
                return pickinfo.pickedPoint;
            }
            return null;
        };
        // 끝쒼渴흙麟깃솰桂（객苟）
        const onPointerDown = (evt: any) => {
            if (evt.button !== 0) { return; }
            const pickInfo = scene.pick(scene.pointerX, scene.pointerY,
                function (mesh) { return mesh !== plan && mesh.isPickable !== false; });
            if (pickInfo.hit) {
                currentMesh = pickInfo.pickedMesh;
                if (currentMesh.name.indexOf('tip') !== -1) {
                    startingPoint = getGroundPosition(evt);
                }
            }
        };
        // 끝쒼渴흙麟깃솰桂（盧땡）
        const onPointerMove = (evt: any) => {
            if (!startingPoint) { return; }
            const current = getGroundPosition(evt);
            if (!current) { return; }
            if (currentMesh.name.indexOf('tip') !== -1) {
                if (currentMesh === this.tipA) {
                    currentMesh.position.x = startingPoint.x;
                    this.a = Math.abs(currentMesh.position.x);
                } else if (currentMesh === this.tipB) {
                    currentMesh.position.y = startingPoint.y;
                    this.b = Math.abs(currentMesh.position.y);
                }
                this.updateLineData(startingPoint, currentMesh);
            }
            startingPoint = current;
        };
        // 끝쒼渴흙麟깃솰桂（偈폅）
        const onPointerUp = () => {
            if (startingPoint) {
                startingPoint = null;
                return;
            }
        };
        canvas.addEventListener('pointerdown', onPointerDown, false);
        canvas.addEventListener('pointerup', onPointerUp, false);
        canvas.addEventListener('pointermove', onPointerMove, false);

        scene.onDispose = function () {
            canvas.removeEventListener('pointerdown', onPointerDown);
            canvas.removeEventListener('pointerup', onPointerUp);
            canvas.removeEventListener('pointermove', onPointerMove);
        };
    }

    /**
     * 뫘劤窟係
     */
    updateLineData(startingPoint: Vector3, currentMesh: BABYLON.AbstractMesh): void {
        this.tipA.position = new Vector3(-this.a, 0, 0);
        this.tipB.position = new Vector3(0, this.b, 0);
        this.tipC.position = new Vector3(0, -this.b, 0);
        this.tipD.position = new Vector3(this.a, 0, 0);

        // this.tipP1.position = new Vector3(-this.a, this.b, 0);

        this.r = Vector3.Distance(new Vector3(this.a, this.b, 0), Vector3.Zero());
        if (this.IsMoved === false) {
            if (startingPoint !== null) {
                if (currentMesh === this.tipP1) {
                    currentMesh.position = startingPoint.scale(1).normalize().scale(this.r);
                } else {
                    this.tipP1.position = this.tipP1.position.scale(1).normalize().scale(this.r);
                }
            }
        } else {
            this.tipP1.position = new Vector3(-this.a, this.b, 0);
        }

        const a = Vector3.Dot(new Vector3(1, 0, 0), this.tipP1.position.scale(1).normalize());
        const xV: number = this.tipP1.position.y > 0 ? Math.acos(a) * 180 / Math.PI : 360 - Math.acos(a) * 180 / Math.PI;

        this.tipP2.position = new Vector3(-this.tipP1.position.x, this.tipP1.position.y, 0);
        this.tipP3.position = new Vector3(-this.tipP2.position.x, -this.tipP2.position.y, 0); // FormulasUtils.GetCirclePoint(this.r, 360 - xV);// new Vector3(-this.a, -this.b, 0);
        this.tipP4.position = new Vector3(-this.tipP1.position.x, -this.tipP1.position.y, 0); // new Vector3(this.a, -this.b, 0);

        this.updateLinesMeshVerticesData(this.Circle, Vector3Utils.ToArray(FormulasUtils.GetCircleUpdateVertices(this.r, 360)));
        this.updateLinesMeshVerticesData(this.Ellipse, Vector3Utils.ToArray(FormulasUtils.GetEllipseVertices(this.a, this.b, 360)));

        this.updateLinesMeshVerticesData(this.LineA, Vector3Utils.ToArray([new Vector3(this.a, this.offset, 0), new Vector3(this.a, -this.offset, 0)]));
        this.updateLinesMeshVerticesData(this.LineB, Vector3Utils.ToArray([new Vector3(-this.a, this.offset, 0), new Vector3(-this.a, -this.offset, 0)]));
        this.updateLinesMeshVerticesData(this.LineC, Vector3Utils.ToArray([new Vector3(this.offset, this.b, 0), new Vector3(-this.offset, this.b, 0)]));
        this.updateLinesMeshVerticesData(this.LineD, Vector3Utils.ToArray([new Vector3(this.offset, -this.b, 0), new Vector3(-this.offset, -this.b, 0)]));

        if (Math.abs(this.tipP1.position.x) === this.a && Math.abs(this.tipP1.position.y) === this.b) {
            this.LineA.isVisible = this.LineB.isVisible = this.LineC.isVisible = this.LineD.isVisible = true;
        } else {
            this.LineA.isVisible = this.LineB.isVisible = this.LineC.isVisible = this.LineD.isVisible = false;
        }
    }
    offset = 20;
    /**
     * 댑갭꼍朧
     * @param i 
     */
    Answer(i: number) {
        if (i === 1) {
            this.IsMoved = true;
            this.tipP1.name = 'ti';
            this.tipP1.position = new Vector3(-this.a, this.b, 0);
        } else if (i === 2) {
            this.tipP1.name = 'tipP1';
            this.IsMoved = false;
        } else if (i === 3) {
            this.tipP1.name = 'tipP1';
            this.IsMoved = false;
        }
        this.updateLineData(null, null);
    }

    /**
     * '路零'객큐
     */
    reset(): void {
        this.Answer(1);
    }
}
